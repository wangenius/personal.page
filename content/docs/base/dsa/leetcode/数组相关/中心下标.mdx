# 寻找数组的中心下标

```cpp
int pivotIndex(vector<int> &nums){
    if (nums.size() == 1)
        return 0;
    int anchor = 0;
    int left = 0, right = -nums[0];
    for (auto i : nums)
        right = right + i;

    if (left == right)
        return anchor;
    while (anchor < nums.size() - 1)
    {
        left = left + nums[anchor];
        anchor++;
        right = right - nums[anchor];
        if (left == right)
            return anchor;
    }

    return -1;
};
```

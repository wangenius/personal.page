---
title: 杨辉三角
---

```cpp
  vector<vector<int>> generate(int numRows) {
    vector<vector<int>> res(numRows);
    for (int i = 0; i < numRows; ++i) {
      res[i].resize(i + 1);
      // 第i+1排
      for (int j = 0; j <= i; ++j)
        res[i][j] = j == 0 || j == i ? 1 : res[i - 1][j - 1] + res[i - 1][j];
    }
    return res;
  };
```
